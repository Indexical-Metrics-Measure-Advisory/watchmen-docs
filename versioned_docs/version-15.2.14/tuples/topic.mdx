export enum TopicKind {
	SYSTEM = 'system',
	BUSINESS = 'business'
}

export enum TopicType {
	RAW = 'raw',
	META = 'meta',
	DISTINCT = 'distinct',
	AGGREGATE = 'aggregate',
	TIME = 'time',
	RATIO = 'ratio'
}

export type TopicId = string;

export interface Topic extends Tuple {
	topicId: TopicId;
	name: string;
	kind: TopicKind;
	type: TopicType;
	description?: string;
	factors: Array<Factor>;
	tenantId?: TenantId;
	dataSourceId?: DataSourceId;
}

	name: string;
	label: string;
	type: FactorType;
	enumId?: EnumId;
	defaultValue?: string;

	description?: string;
	createTime: DateTime;
	lastModified: DateTime;
}
